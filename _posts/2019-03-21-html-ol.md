```html
<html lang="ko">
<!--oltest.html --> 
<head>
    <title>ol 태그</title>
    <meta charset="UTF-8">
</head>
<body>
    <h1>ol 태그 </h1>
    <ol>
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <h1>시작번호 지정 : start </h1>
    <ol start="50">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <h1>특정 목록의 번호 지정 : value </h1>
    <ol start="50">
      <li>Coffee</li>
      <li value="60">Tea</li>
      <li>Milk</li>
    </ol>
    <h1>type 지정 : 1 | A | a | I | i </h1>
    <ol type="1">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <ol type="A">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <ol type="a">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <ol type="I">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
    <ol type="i">
      <li>Coffee</li>
      <li>Tea</li>
      <li>Milk</li>
    </ol>
</body>
</html>
```
