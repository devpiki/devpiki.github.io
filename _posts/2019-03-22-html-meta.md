```html
<!DOCTYPE html>
<html>
<head>
	<meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
	<meta name="Date" content="2016-02-15T07:45:37+09:00" />

	<meta http-equiv="X-UA-Compatible" content="IE=edge" />
	<!-- 브라우저 호환성 지정 -->

	<!--<meta http-equiv="Refresh" content="60" /> -->
	<!-- 화면 갱신 / 60초-->
	<!-- <meta http-equiv="refresh" content="3;url=https://www.daum.net/"> -->
	<meta http-equiv="Cache-Control" content="no-cache" /> 
	<meta http-equiv="Pragma" content="no-cache" />
	<!--캐시가 되지 않도록 지정-->
	<meta http-equiv="Expires" content="Mon, 08 Sep 2011 12:00:00 GMT"> 
	<!-- 캐쉬 만료일 -->

	<meta property="og:url" content="https://devpiki.github.io/">
	<meta property="og:type" content="website">
	<meta property="og:title" content="Piki's blog">
	<meta property="og:image" content="//devpiki.github.io/assets/smiley.gif">
	<meta property="og:description" content="html5, css3, javascript">

	<title></title>

</head>
<body>

</body>
</html>
```
